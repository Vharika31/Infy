
#include<bits/stdc++.h>
using namespace std;

int main(){
    int N;
    cin>>N;
    int cnt = 0;
    for(int i = 1;i<=N;i++){
        if(N%i == 0)
        cnt++;
    }
    if(cnt == 2)
    cout<<"Prime";
    else
        cout<<"Not prime";

}

Time Complexity: O(N)
